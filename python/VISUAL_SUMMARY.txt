│ 3. API USAGE PATTERN                                                  │
└──────────────────────────────────────────────────────────────────────┘

  BEFORE (Old Pattern):
  ────────────────────
    t = 0.0
    nt = 0
    while t < Tf:
        MRadaptation(mra_config)
        t += dt
        nt += 1
        print(f"iteration {nt}: t = {t:.4f}, cells = {mesh.nb_cells}")
        # ... time step ...

  AFTER (New Pattern):
  ───────────────────
    with progress.time_loop(Tf, dt, desc="Simulation") as pbar:
        while True:
            with pbar.mesh_adaptation(mesh):
                MRadaptation(mra_config)
            if not pbar.advance(dt, mesh=mesh):
                break
            # ... time step ...
